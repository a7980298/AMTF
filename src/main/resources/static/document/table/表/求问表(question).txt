CREATE TABLE AMTF_Question(
      --id
      Question_id number(20) not null,
 
      --内容
      Question_Text varchar2(200),
      
      --时间
      Question_Date Date,
      
      --User账户id
      User_id number(20)     
  )


    
create sequence AMTF_Question_id start with 1 increment by 1;
    

create or replace trigger AMTF_AMTF_Question
before insert on AMTF_Question    
for each row  
begin  
    select AMTF_Question_id.nextval into :new.Question_id from dual; 
end ;
