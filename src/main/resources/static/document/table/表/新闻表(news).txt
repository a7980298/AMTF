CREAT TABLE AMTF_News(
      --id
      News_id number(20) not null,

      --标题
      News_Head varchar2(50),

      --内容
      News_Text varchar2(2000),

      --照片
      News_Img varchar2(20),

      --时间
      News_Date Date,

      --User账户id
      User_id number(20)     
  )


create sequence AMTF_News_id start with 1 increment by 1;
    
create or replace trigger AMTF_AMTF_News
before insert on AMTF_News    
for each row  
begin  
    select AMTF_News_id.nextval into :new.News_id from dual; 
end ;
