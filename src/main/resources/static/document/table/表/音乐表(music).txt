CREATE TABLE AMTF_Music(
      --id
      Music_id number(20) not null,
      
      --音乐标题
      Music_Head varchar(20),
      
      --音乐
      Music_Path varchar2(50),
      
      --时间
      Music_Date Date,
      
      --User账户id
      User_id number(20)     
  )


create sequence AMTF_Music_id start with 1 increment by 1;
    
create or replace trigger AMTF_AMTF_Music
before insert on AMTF_Music    
for each row  
begin  
    select AMTF_Music_id.nextval into :new.Music_id from dual; 
end ;
