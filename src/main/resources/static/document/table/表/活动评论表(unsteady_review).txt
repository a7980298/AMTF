CREATE TABLE AMTF_Unsteady_Review(
      --id
      Unsteady_Review_id number(20) not null,
 
      --活动id
      Unsteady_id number(20),

      --内容
      Unsteady_Review_Text varchar2(100),
      
      --时间
      Unsteady_Review_Date Date,
      
      --User账户id
      User_id number(20)     
  )



create sequence AMTF_Unsteady_Review_id start with 1 increment by 1;
    

create or replace trigger AMTF_AMTF_Unsteady_Review 
before insert on AMTF_Unsteady_Review  
for each row  
begin  
    select AMTF_Unsteady_Review_id  .nextval into :new.Unsteady_Review_id   from dual; 
end ;
