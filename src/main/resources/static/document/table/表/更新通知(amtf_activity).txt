CREATE TABLE amtf_activity(
	activity_id INT NOT NULL, -- id
	activity_name VARCHAR(30) NOT NULL, -- 管理员姓名
	activity_head VARCHAR(30), -- 活动标题
	activity_check CHAR(1), -- 参与级别
	activity_sttymd INT, -- 开始日期
	activity_endymd INT, -- 结束日期
	activity_text VARCHAR(200), -- 活动内容
	activity_time DATETIME	-- 发布时间
)
